# create python virtual environment
python -m venv venv

# lets PowerShell run scripts only in the current session without permanently lowering security
# by default PowerShell blocks scripts to protect the system
Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope Process

# activate virtual environment
.\venv\Scripts\Activate.ps1

# install required packages
pip install selenium pytest allure-pytest loguru requests urllib3 webdriver-manager pyautogui pynput

"""
selenium: automate browser actions for testing web apps

pytest: runs tests with powerful features and plugins

allure-pytest: generates detailed test reports integrated with pytest

requests: simplifies HTTP requests for API testing or web interactions

urllib3: handles HTTP connections, used by requests internally

webdriver-manager: automatically download and manage browser drivers

pyautogui: automate keyboard and mouse actions outside the browser, like file uploads, alerts

pynput: listens to and controls input devices (keyboard/mouse) to simulate more complex input behavior
"""

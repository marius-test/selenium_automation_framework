# File: pytest_help.txt
# Purpose: Run tests from the PowerShell console.

# To run a single test file:
pytest tests/test_login.py -v --html=report.html

# To run all tests in the /tests folder:
pytest tests/ -v --html=report.html

# To run a single test function from a test file:
pytest tests/test_login.py::test_successful_login_variants -v --html=report.html

# To run one test for debugging:
# Add the test code inside the debug_test.py script and run it.
python debug/debug_test.py

# -v enables verbose output.
# --html=report.html generates an HTML report (requires pytest-html package).
